Scene: Game
- Pause

game -> pause: 
	GameManager.Pause()
	UIManager.Pause()
pause -> game:
	UIManager.Resume()
	GameManager.Resume()
pause -> settings:
	UIManager.Resume()
	UIManager.OpenSettings()
settings -> pause:
	UIManager.CloseSettings()
	UIManager.Pause()


mainmenu -> levelselector:
	UIManager.CloseMM()
	UIManager.OpenSelector()
levelselector -> mainmenu:
	UIManager.CloseSelector()
	UIManager.OpenMM()
mainmenu -> settings:
	UIManager.CloseMM()
	UIManager.OpenSettings()
settings -> mainmenu
	UIManager.CloseSettings()
	UIManager.OpenMM
